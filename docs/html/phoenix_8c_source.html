99 void phoenixSetQEBit()
100 {
101  uint8_t SRArray[2] = {0, 0};
102  // Read both status register bytes.
103 #if (PARTNO == AT25DL081) || \
104  (PARTNO == AT25DL161) || \
105  (PARTNO == AT25DF081A) || \
106  (PARTNO == AT25DF321A) || \
107  (PARTNO == AT25DF641A)
108  phoenixReadSR(SRArray);
109 #else
110  SRArray[0] = phoenixReadSRB1();
111  SRArray[1] = phoenixReadSRB2();
112 #endif
113  // Store a 1 in the QE bit.
114  SRArray[1] |= (1 << 1);
115  // Altering the device data, so send over a write enable first.
116  phoenixWriteEnable();
117  // Store the command and new SRB values. NOTE this needs to be done after
118  // the write enable since the internal buffer used is a shared resource.
119 #if (PARTNO == AT25DL081) || \
120  (PARTNO == AT25DL161) || \
121  (PARTNO == AT25DF081A) || \
122  (PARTNO == AT25DF321A) || \
123  (PARTNO == AT25DF641A)
124  phoenixWriteSRB1(SRArray[0]);
125  phoenixWriteSRB2(SRArray[1]);
126 #else
127  phoenixWriteSR(SRArray, 2);
128 #endif
129  if(DISPLAY_OUTPUT)
130  {
131  printSPIExchange(txPhoenixInternalBuffer, 3, NULL, 0);
132  }

501 void phoenixDualIORead(uint32_t address, uint8_t *rxBuffer, uint32_t rxNumBytes, uint8_t readMode, uint8_t modeByteValue)
502 {
503  // Set to an unused number to ensure a transmission does not take place with invalid input.
504  uint8_t transmissionSelect = 0x02;
505  switch(readMode)
506  {
507  // Single Dual Read
508  case 0:
509  load4BytesToTxBuffer(txPhoenixInternalBuffer, CMD_PHOENIX_DUAL_IO_READ, address);
510  txPhoenixInternalBuffer[4] = 0x00;
511  transmissionSelect = 0x00;
512  break;
513  // Start Continuous Dual Read
514  case 1:
515  load4BytesToTxBuffer(txPhoenixInternalBuffer, CMD_PHOENIX_DUAL_IO_READ, address);
516  txPhoenixInternalBuffer[4] = modeByteValue;
517  transmissionSelect = 0x00;
518  break;
519  // Continue Continuous Dual Read
520  case 2:
521  txPhoenixInternalBuffer[0] = (uint8_t) (address >> 16);
522  txPhoenixInternalBuffer[1] = (uint8_t) (address >> 8);
523  txPhoenixInternalBuffer[2] = (uint8_t) address;
524  txPhoenixInternalBuffer[3] = modeByteValue;
525  transmissionSelect = 0x01;
526  break;
527  // End Continuous Dual Read
528  case 3:
529  txPhoenixInternalBuffer[0] = (uint8_t) (address >> 16);
530  txPhoenixInternalBuffer[1] = (uint8_t) (address >> 8);
531  txPhoenixInternalBuffer[2] = (uint8_t) address;
532  txPhoenixInternalBuffer[3] = 0x00;
533  transmissionSelect = 0x01;
534  break;
535  default:
536  printf("Error with phoenixDualIORead.\n");
537  printf("\t- Invalid dualReadMode value. Expected 0-3, received %d.\n", readMode);
538  printf("\tSee documentation for operating modes.\n");
539  break;
540  }
541 
542  if(transmissionSelect == 0x00)
543  {
544  SPI_DualExchange(1, txPhoenixInternalBuffer, 5, rxBuffer, rxNumBytes, 0);
545  if(DISPLAY_OUTPUT)
546  {
547  printSPIExchange(txPhoenixInternalBuffer, 5, rxBuffer, rxNumBytes);
548  }
549  }
550  else if(transmissionSelect == 0x01)
551  {
552  SPI_DualExchange(0, txPhoenixInternalBuffer, 4, rxBuffer, rxNumBytes, 0);
553  if(DISPLAY_OUTPUT)
554  {
555  printSPIExchange(txPhoenixInternalBuffer, 4, rxBuffer, rxNumBytes);
556  }
557  }
558 }
