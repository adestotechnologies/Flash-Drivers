353 uint32_t fusionTest()
354 {
355  // Sets the various pins as inputs and output
356  SPI_ConfigureSingleSPIIOs();
357 
358  // Instantiate the arrays needed for testing purposes.
359  // dataWrite is used when sending data that will be flashed to the device.
360  uint8_t dataWrite[MAXIMUM_BUFFER_SIZE] = {0};
361  // dataRead is used as a buffer for received data. Read data will be stored here.
362  uint8_t dataRead[MAXIMUM_BUFFER_SIZE] = {0};
363  // dataTest is used as a buffer for comparisons. dataRead will be
364  // loaded with data and compared against this buffer for equality.
365  uint8_t dataTest[MAXIMUM_BUFFER_SIZE] = {0};
366  // Count the number of errors, this is output at the end of the testbench.
367  uint32_t errorCount = 0;
368 
369  // Store all Fs in the test buffer for later comparison
370  for(int i = 0; i < MAXIMUM_BUFFER_SIZE; i++)
371  {
372  dataTest[i] = 0xFF;
373  }
374 #if PARTNO == AT25XE512C
375  // Store the manufacturing ID for later comparison
376  uint8_t MID[4] = {0x1F, 0x65, 0x01, 0x00};
377 #elif PARTNO == AT25XE011
378  // Store the manufacturing ID for later comparison
379  uint8_t MID[4] = {0x1F, 0x42, 0x00, 0x00};
380 #elif PARTNO == AT25XE021A
381  // Store the manufacturing ID for later comparison
382  uint8_t MID[4] = {0x1F, 0x43, 0x01, 0x00};
383 #elif PARTNO == AT25XE041B
384  // Store the manufacturing ID for later comparison
385  uint8_t MID[4] = {0x1F, 0x44, 0x02, 0x00};
386 #elif PARTNO == AT25DN256
387  // Store the manufacturing ID for later comparison
388  uint8_t MID[4] = {0x1F, 0x40, 0x00, 0x00};
389 #elif PARTNO == AT25DN512C
390  // Store the manufacturing ID for later comparison
391  uint8_t MID[4] = {0x1F, 0x65, 0x01, 0x00};
392 #elif PARTNO == AT25DN011
393  // Store the manufacturing ID for later comparison
394  uint8_t MID[4] = {0x1F, 0x42, 0x00, 0x00};
395 #elif PARTNO == AT25DF256
396  // Store the manufacturing ID for later comparison
397  uint8_t MID[4] = {0x1F, 0x40, 0x00, 0x00};
398 #elif PARTNO == AT25DF512C
399  // Store the manufacturing ID for later comparison
400  uint8_t MID[4] = {0x1F, 0x64, 0x01, 0x00};
401 #elif PARTNO == AT25DF011
402  // Store the manufacturing ID for later comparison
403  uint8_t MID[4] = {0x1F, 0x42, 0x00, 0x00};
404 #elif PARTNO == AT25DF021A
405  // Store the manufacturing ID for later comparison
406  uint8_t MID[4] = {0x1F, 0x43, 0x01, 0x00};
407 #elif PARTNO == AT25DF041B
408  // Store the manufacturing ID for later comparison
409  uint8_t MID[4] = {0x1F, 0x44, 0x02, 0x00};
410 #elif PARTNO == AT25XV021A
411  // Store the manufacturing ID for later comparison
412  uint8_t MID[4] = {0x1F, 0x43, 0x01, 0x00};
413 #elif PARTNO == AT25XV041B
414  // Store the manufacturing ID for later comparison
415  uint8_t MID[4] = {0x1F, 0x44, 0x02, 0x00};
416 #endif
417 
418  // Load up dataWrite with a pattern that will be stored in flash memory.
419  for(uint32_t a = 0; a < 100; a++)
420  {
421  dataWrite[a] = (uint8_t) 0xA+a;
422  }
423 
424  /********************************************************************
425  * 1. Read manufacturing ID, test write enable, test write disable. *
426  ********************************************************************/
427 
428  printf("\n\nTesting Read MID, WEnable, and WDisable ----------\n\n");
429 
430  // Part A: Test that the MID can be read.
431  // Read the manufacturing ID and store the returned data in dataRead
432  fusionReadMID(dataRead);
433  // Compare the read data to the expected bytes stored in MID. Print
434  // the result of the test.
435  if(compareByteArrays(dataRead, MID, 4))
436  {
437  printf("ReadMID Success.\n");
438  }
439  else
440  {
441  printf("ReadMID fail.\n");
442  errorCount++;
443  }
444 
445 
446  // Part B: Ensure that the device can be write enabled.
447  // The device is write enabled...
448  fusionWriteEnable();
449  // ... the status register is then read ..
450  fusionReadSR(dataRead);
451  // ... and the write enable bit checked.
452  // WE should be 1 if the command and setup are operational.
453  if(!((dataRead[0] >> 1) & 1))
454  {
455  printf("Error with WE or Read SR");
456  errorCount++;
457  }
458  else
459  {
460  printf("WE success.\n");
461  }
462 
463 
464  // Part C: Now attempt to disable write.
465  // The command is sent to disable write...
466  fusionWriteDisable();
467  // ... the status register is then read...
468  fusionReadSR(dataRead);
469  // ... and the write enable bit checked.
470  // WE should be 0 if the command and setup are operational.
471  if(((dataRead[0] >> 1) & 1))
472  {
473  printf("Error with WD or Read SR");
474  errorCount++;
475  }
476  else
477  {
478  printf("WD success.\n");
479  }
480 
481 #if (PARTNO == AT25XE021A) || \
482  (PARTNO == AT25XE041B) || \
483  (PARTNO == AT25DF021A) || \
484  (PARTNO == AT25DF041B) || \
485  (PARTNO == AT25XV021A) || \
486  (PARTNO == AT25XV041B) || \
487  (ALL == 1)
488  /********************************************************************
489  * 2. Test that sectors can be properly unprotected. *
490  ********************************************************************/
491 
492 
493  printf("\n\nTesting Un/protect Sector ------------------------\n\n");
494 
495  // Part A: Guarantee sector 0 is protected.
496  // To start with, ensure that sector 0 is protected. It should be if
497  // the test bench has been run before this and completed.
498  fusionReadSectorProtectionRegisters(0, dataRead);
499  // The register should read 0xFF is protected.
500  if(dataRead[0] != 0xFF)
501  {
502  printf("Error with protection SR of sector 0");
503  errorCount++;
504  }
505  else
506  {
507  printf("Protect success.\n");
508  }
509 
510  // Part B: Unprotect sector 0.
511  // Write enable the device since we'll be writing to it...
512  fusionWriteEnable();
513  // ... and unprotect sector 0.
514  fusionUnprotectSector(0);
515  // Follow up by reading the status register which should be 0x00
516  // if the sector was unprotected properly.
517  fusionReadSectorProtectionRegisters(0, dataRead);
518  // Sector 0 should be unprotected (0x00)
519  if(dataRead[0] != 0x00)
520  {
521  printf("Error unprotecting sector 0");
522  errorCount++;
523  }
524  else
525  {
526  printf("Unprotect success.\n");
527  }
528 #endif
529  /********************************************************************
530  * 3. Test the page erase functionality. *
531  ********************************************************************/
532 
533  printf("\n\nTesting Erase ------------------------------------\n\n");
534 
535  // Test by erasing the device and confirming all bits are set.
536  // Write enable the device since we'll be issuing a command that will modify stored data.
537  fusionWriteEnable();
538  // Issue a page erase command starting from address 0 (page 0).
539  fusionPageErase(0);
540  // Wait for the operation to complete.
541  fusionWaitOnReady();
542  // Now read the flash device. All bits should be set -> FF for each byte.
543  fusionReadArray(0, dataRead, 100);
544  // If any of the first 100 bytes are not 0xFF, print an error message.
545  if(compareByteArrays(dataRead, dataTest, 100))
546  {
547  printf("PageErase success.\n");
548  }
549  else
550  {
551  printf("PageErase fail.\n");
552  errorCount++;
553  }
554 
555 #if (PARTNO == AT25XE021A) || \
556  (PARTNO == AT25XE041B) || \
557  (PARTNO == AT25DF021A) || \
558  (PARTNO == AT25DF041B) || \
559  (PARTNO == AT25XV021A) || \
560  (PARTNO == AT25XV041B) || \
561  (ALL == 1)
562  /********************************************************************
563  * 4. Test the sequential/byte program mode. *
564  ********************************************************************/
565 
566  printf("\n\nTesting Sequential Program Mode -------------------\n\n");
567 
568  // Part A: Sequentially program the device to confirm sequential program works.
569  // Write enable the device since we'll be issuing a command that will modify stored data.
570  fusionWriteEnable();
571  // Begin sequentially programming the device byte by byte from address 0.
572  // The first byte programmed is included in the initial program command.
573  fusionSequentialProgramModeEnable(0, 0);
574  // Set the first element of the comparison buffer to zero, the byte programmed above.
575  dataTest[0] = 0x00;
576  // Now program 48 other bytes, and store the values in the test buffer. Note the offset
577  // (j=1) since the first byte was already programmed.
578  for(int j = 1; j < 50; j++)
579  {
580  dataTest[j] = (uint8_t) j;
581  fusionSequentialProgramMode(j);
582  }
583 
584  // Part B: Leave sequential program mode.
585  // Exit sequential program mode by issuing a write disable command.
586  fusionWriteDisable();
587 
588  // Part C: Test that the data was written properly.
589  // For the sake of testing that we can no longer write in sequential program mode, try
590  // writing one last byte. Since we erase this memory location in a prior operation, a
591  // 'failed' write in sequential mode means it should still be FF. FF is stored in the buffer
592  // for this element so an error will be printed if they do not match.
593  fusionSequentialProgramMode(0x0F);
594  // Read the array, and compare against known data.
595  fusionReadArray(0, dataRead, 100);
596  if(compareByteArrays(dataRead, dataTest, 51))
597  {
598  printf("Sequential operation success.\n");
599  }
600  else
601  {
602  printf("Sequential operation fail.\n");
603  errorCount++;
604  }
605  // We checked the 'failed' sequential program mode above, but if indeed it still programmed,
606  // This will print out a directed message.
607  if(dataRead[0x50] != 0xFF)
608  {
609  printf("Error, data programmed when write disabled.");
610  errorCount++;
611  }
612 
613  /********************************************************************
614  * 5. Test dual read and write. *
615  ********************************************************************/
616 
617  printf("\n\nTesting Dual Read/Write Mode -------------------------\n\n");
618 
619  // Part A: Program the device in dual mode.
620  // Write enable the device since we'll be modifying data.
621  fusionWriteEnable();
622  // Perform a block erase to setup the test.
623  fusionBlockErase4K(0);
624  // Wait for the block erase to finish.
625  fusionWaitOnReady();
626  // Reissue the write enable command since we'll be modifying the data a second time.
627  fusionWriteEnable();
628  // Program the first 50 bytes in dual input program mode.
629  fusionDualInputProgram(0, dataWrite, 50);
630  // Wait for the operation to complete. (data is sent to a buffer and the write to flash occurs
631  // after CSb is deselected).
632  fusionWaitOnReady();
633 
634  // Part B: Confirm dual mode write worked.
635  // Now perform a standard read. This ensures that the data is readable in both formats
636  // and that there wasn't some sort of odd segmentation.
637  fusionReadArray(0, dataRead, 100);
638  // Compare against known data.
639  if(compareByteArrays(dataRead, dataWrite, 50))
640  {
641  printf("Dual input program operation success.\n");
642  }
643  else
644  {
645  printf("Dual input program operation fail.\n");
646  errorCount++;
647  }
648 
649  // Part C: Test that dual output read works.
650  // Now test dual output read...
651  fusionDualOutputRead(0, dataRead, 100);
652  // ... and compare against known data.
653  if(compareByteArrays(dataRead, dataWrite, 50))
654  {
